# ========== Room Transfer 控制面 ==========

# 预览只校验输入，不调用服务：
node tools/rollout/rollout-control-plane-cli.js ^
  --dry-run ^
  --world-id local ^
  --rollout-epoch rollout-20260722-a ^
  --room-id room-empty-001 ^
  --old-server-id game-server-001 ^
  --new-server-id game-server-002 ^
  --proxy-instance-id game-proxy-001 ^
  --backup-reference backup-room-empty-001 ^
  --reason "controlled empty-room transfer"

# 预检和执行只能通过 admin-api JWT。执行会先取得预检 nonce，再提交 nonce 绑定的请求；
# admin-api 从 registry 解析 old/new/proxy endpoint，并为每个写步骤签发短时断言。
node tools/rollout/rollout-control-plane-cli.js ^
  --admin-api-url http://127.0.0.1:3001 ^
  --admin-api-token "%ADMIN_API_TOKEN%" ^
  --execute ^
  --world-id local ^
  --rollout-epoch rollout-20260722-a ^
  --room-id room-empty-001 ^
  --old-server-id game-server-001 ^
  --new-server-id game-server-002 ^
  --proxy-instance-id game-proxy-001 ^
  --backup-reference backup-room-empty-001 ^
  --reason "controlled empty-room transfer"

# PowerShell 包装入口只会调用同一控制面 CLI：
powershell -ExecutionPolicy Bypass -File scripts/ops/rollout-three-process-drill.ps1 ^
  -WorldId local ^
  -RolloutEpoch rollout-20260722-a ^
  -RoomId room-empty-001 ^
  -OldServerId game-server-001 ^
  -NewServerId game-server-002 ^
  -ProxyInstanceId game-proxy-001
